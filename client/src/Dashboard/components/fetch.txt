   useEffect(() => {
            fetch('/auth/getResultatPrinc')

              if (data.resultat_principale && Array.isArray(data.resultat_principale)) {
                setData(data.resultat_principale);
                setRecord(data.resultat_principale);

              } else {

            })
            .catch((error) => {

            });
      
    }, []);
